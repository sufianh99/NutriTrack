    5. Add three new route functions AFTER the existing dashboard route:

    ```python
    @bp.route("/food/add", methods=["GET", "POST"])
    def add_food():
        profile = _get_profile()
        if profile is None:
            return redirect(url_for("main.onboarding"))
        form = FoodEntryForm()
        if form.validate_on_submit():
            entry = FoodEntry(
                date=date.today(),
                name=form.name.data,
                amount_g=form.amount_g.data,
                calories_per_100g=form.calories_per_100g.data,
                protein_per_100g=form.protein_per_100g.data,
                fat_per_100g=form.fat_per_100g.data,
                carbs_per_100g=form.carbs_per_100g.data,
            )
            db.session.add(entry)
            db.session.commit()
            flash("Lebensmittel hinzugefuegt.", "success")
            return redirect(url_for("main.dashboard"))
        return render_template("food_form.html", form=form, edit=False)


    @bp.route("/food/<int:entry_id>/edit", methods=["GET", "POST"])
    def edit_food(entry_id: int):
        profile = _get_profile()
        if profile is None:
            return redirect(url_for("main.onboarding"))
        entry = db.session.get(FoodEntry, entry_id)
        if entry is None:
            flash("Eintrag nicht gefunden.", "warning")
            return redirect(url_for("main.dashboard"))
        form = FoodEntryForm(obj=entry)
        if form.validate_on_submit():
            form.populate_obj(entry)
            db.session.commit()
            flash("Eintrag aktualisiert.", "success")
            return redirect(url_for("main.dashboard"))
        return render_template("food_form.html", form=form, edit=True, entry=entry)


    @bp.route("/food/<int:entry_id>/delete", methods=["POST"])
    def delete_food(entry_id: int):
        entry = db.session.get(FoodEntry, entry_id)
        if entry is not None:
            db.session.delete(entry)
            db.session.commit()
            flash("Eintrag geloescht.", "success")
        return redirect(url_for("main.dashboard"))
    ```

    Key design decisions:
    - `add_food` supports GET (show empty form) and POST (process submission) -- per FOOD-01
    - `edit_food` uses `FoodEntryForm(obj=entry)` to pre-fill and `form.populate_obj(entry)` to update -- per FOOD-04
    - `delete_food` is POST-only (no GET) to prevent accidental deletion -- per FOOD-05 and research anti-patterns
    - `db.session.get(FoodEntry, entry_id)` used instead of deprecated `Model.query.get()` -- per SQLAlchemy 2.x convention
    - Flash messages in German, no special characters (use "ue" for umlauts in flash messages to avoid encoding issues in tests)
    - All routes check `_get_profile()` and redirect to onboarding if None (except delete which just redirects to dashboard)
